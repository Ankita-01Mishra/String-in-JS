8. Palindrome Check

        var Palin_Check = (str) =>
        {
          let rev = "";
          for (let i = str.length - 1; i >= 0; i--) {
              rev += str[i];
          }
          if(rev===str){
            return "True"
          }
          return "False";
         }

9. Reverse the String

        var Reverse_String = (str) => 
        {
            let rev = "";
            for (let i = str.length - 1; i >= 0; i--) {

            }
            return rev;
        };
